101  public IPatternIterator GetPatternIterator<T>(bool isReversed = false) where T : IPatternIterator
102  {

171  private void BuildShape()
172  {
173  m_shapeImplementation = null;
174 
175  switch (m_shapePattern.currentGeometry)
176  {
177  case ShapePattern.Geometry.CIRCLE:
178  {
179  m_shapeImplementation = new CircleImplementation();
180  break;
181  }
182  case ShapePattern.Geometry.LINE:
183  {
184  m_shapeImplementation = new LineImplementation();
185  m_shapePattern.numberOfPoints = 2;
186  break;
187  }
188  case ShapePattern.Geometry.ARC:
189  {
190  m_shapeImplementation = new ArcImplementation();
191  break;
192  }
193  case ShapePattern.Geometry.N_SIDED:
194  {
195  m_shapeImplementation = new NSidedImplementation();
196  if (m_shapePattern.numberOfPoints < 3)
197  {
198  m_shapePattern.numberOfPoints = 3;
199  }
200  break;
201  }
202  case ShapePattern.Geometry.N_SIDED_COMPLEX:
203  {
204  m_shapeImplementation = new ComplexNSidedImplementation();
205  if (m_shapePattern.numberOfPoints < 6)
206  {
207  m_shapePattern.numberOfPoints = 6;
208  }
209  m_shapePattern.spreadInDegrees = 360;
210  break;
211  }
212  }
213 
214  if (m_shapeImplementation != null)
215  {
216  m_shapeImplementation.Build(transform.position, m_shapePattern, ref m_shapePoints);
217  Vector2 position = transform.position;
218  m_forwardVector = MathHelper.GetNormalizedVectorFromRotationInDegrees(m_shapePattern.forwardRotationInDegrees) * (m_shapePattern.maxRadius);
219  int numberOfPoints = m_shapePoints.Length;
220  m_onPatternRebuild?.Invoke(m_shapePoints);
221  m_patternHasBeenBuilt = true;
222  }
